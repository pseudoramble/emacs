;; blogging.el ;; ------------------ ;; A set of small functions used to help me write & publish blog entries ;; ;; Author: Doug Swain (dswain@gmail.com)

(setq blog-entry-dir (expand-file-name "~/Programming/blog-revamp/entry/docs/")) (setq blog-file-type ".md")

(defun blog-new-entry (name) "Creates a new file under (concat blog-entry-dir name blog-file-type). It does not save it by default" (interactive "sName: ")

(let ((filename (concat blog-entry-dir name blog-file-type))) (let ((blog-entry-buffer (find-file-noselect filename))) (switch-to-buffer blog-entry-buffer) (set-buffer blog-entry-buffer))))

(defun blog-gen-entry () "Takes file and generates HTML from Markdown. Ships it to another tool for additional mods & formatting" (interactive) (let ((filename (file-name-sans-extension (buffer-name (current-buffer))))) (markdown-export "temp.html") (message "%s" filename) ;(shell-command (concat "python generate_entry.py temp.html " filename)) ) )
